export const bundles = BundleTable.decorate(manifest.files, frontMatter.bundles)

## Bundles

We distribute several CSS files, each of which "bundles"
different parts of the system so that you can tailor it to your
site without the need for a custom build process:

<BundleTable files={bundles.filter(file => file.path.endsWith('.css'))}  />

### All CSS and custom fonts

<CodeBlock language='html'>
{`<link rel="stylesheet" href="${publishedUrl('css/sfds.css')}">
<link rel="stylesheet" href="${publishedUrl('css/fonts.css')}">`}
</CodeBlock>

### Only utilities

<CodeBlock language='html'>
{`<link rel="stylesheet" href="${publishedUrl('css/utilities.css')}">`}
</CodeBlock>

:::tip

<div>Replace <code>{defaultPackage.version}</code> in the above link <code>href</code> attributes with <code>latest</code> to get the latest version.</div>

:::

[tailwind]: https://tailwindcss.com/
